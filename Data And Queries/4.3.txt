DELIMITER $
CREATE TRIGGER er4_3 
BEFORE UPDATE ON customer 
FOR EACH ROW 
BEGIN 
IF NEW.email<>OLD.email THEN 
SET NEW.email=OLD.email;
SIGNAL SQLSTATE VALUE '45000'
SET MESSAGE_TEXT = 'YOU CANNOT CHANGE YOUR EMAIL ADDRESS'; 
END IF;
END$
DELIMITER ;
